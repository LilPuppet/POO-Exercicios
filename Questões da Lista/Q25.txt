modelo Livro

inicio do modelo
    inicio dos dados
        texto titulo
        texto autor
        texto editora
        data dataEmprestimo
        data dataDevolucao
        boolean disponivel
    fim dos dados

operação emprestar()
    inicio
        se disponivel = true
            disponivel =false
            set dataEmprestimo
            set dataDevolucao
        senao
            imprime "Este livro não está disponível"
    fim

operação devolver()
    inicio
        se disponivel = false
            disponivel = true
            free dataEmprestimo
            free dataDevolucao
        senao
            imprime "Este livro não foi emprestado"
    fim

fim do modelo
