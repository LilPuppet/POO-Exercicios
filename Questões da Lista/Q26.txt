modelo EquacaoSegundoGrau

inicio do modelo
    inicio dos dados
        float a
        float b
        float c
        float delta
        float bhaskaraMais
        float bhaskaraMenos
    fim dos dados

operação setDelta()
    inicio
        delta = pow(b,2)-4*a*c
    fim

operação setBhaskara()
    inicio
        bhaskaraMais = (-b + pow(delta))/2*a
        bhaskaraMenos = (-b - pow(delta))/2*a
    fim

fim do modelo
